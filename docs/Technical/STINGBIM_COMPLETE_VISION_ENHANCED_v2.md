    /// <summary>

    public class CVEngine

        private YOLOv9Model wallDetector;

        private LayoutLMModel documentUnderstanding;

        public CVEngine()
        {
            LoadModels();
        }
        
        private void LoadModels()
        {
            // YOLOv9 for object detection
            wallDetector = new YOLOv9Model(
                modelPath: "./models/yolov9-wall-detector.onnx",
                confidenceThreshold: 0.85
            );
            
            // SAM for segmentation
            segmentationModel = new SAMModel(
                modelPath: "./models/sam-vit-h.onnx"
            );
            
            // LayoutLMv3 for document understanding
            documentUnderstanding = new LayoutLMModel(
                modelPath: "./models/layoutlmv3-base.onnx"
            );
            
            // Tesseract OCR
            ocrEngine = new TesseractOCR(
                dataPath: "./tessdata",
                language: "eng+ara+fra"  // Multi-language support
            );
        }
        
        public async Task<List<WallDetection>> DetectWalls(Bitmap image)
        {
            // Run YOLOv9 inference
            var detections = await wallDetector.DetectAsync(image);
            
            // Filter by confidence
            var walls = detections
                .Where(d => d.Class == "wall" && d.Confidence > 0.85)
                .Select(d => new WallDetection
                {
                    BoundingBox = d.Box,
                    Confidence = d.Confidence,
                    Type = ClassifyWallType(d)  // Secondary classification
                })
                .ToList();
            
            return walls;
        }
        
        public async Task<List<TextAnnotation>> ExtractText(Bitmap image)
        {
            // Use LayoutLM for document structure understanding
            var layout = await documentUnderstanding.AnalyzeAsync(image);
            
            // Extract text with OCR
            var textRegions = await ocrEngine.RecognizeAsync(image);
            
            // Combine results
            var annotations = textRegions
                .Select(tr => new TextAnnotation
                {
                    Text = tr.Text,
                    BoundingBox = tr.Box,
                    Confidence = tr.Confidence,
                    Type = ClassifyAnnotationType(tr, layout)
                })
                .ToList();
            
            return annotations;
        }
        
        public async Task<RoomBoundary> DetectRoomBoundary(Bitmap floorPlan)
        {
            // Use SAM for precise segmentation
            var segments = await segmentationModel.SegmentAsync(floorPlan);
            
            // Identify room boundary segments
            var roomSegment = segments.FirstOrDefault(s => s.Label == "room");
            
            if (roomSegment != null)
            {
                // Convert segment to polyline
                var boundary = ConvertSegmentToPolyline(roomSegment);
                
                return new RoomBoundary
                {
                    Polyline = boundary,
                    Area = CalculateArea(boundary),
                    Perimeter = CalculatePerimeter(boundary)
                };
            }
            
            return null;
        }

        {
            // Create prediction engine
            var predictionEngine = mlContext.Model.CreatePredictionEngine<
                ProjectFeatures,

            // Make prediction
            var prediction = predictionEngine.Predict(features);
            
            return prediction.EstimatedCost;
        }
        
        public string ClassifyElement(ElementFeatures features)
        {
            var predictionEngine = mlContext.Model.CreatePredictionEngine<
                ElementFeatures,
                ElementClassification>(elementClassificationModel);
            
            var prediction = predictionEngine.Predict(features);
            
            return prediction.Category;
        }
        
        public async Task TrainCostModel(List<HistoricalProject> trainingData)
        {
            // Offline training pipeline

                    "Features",

                    nameof(HistoricalProject.Floors),
                    nameof(HistoricalProject.Complexity))
                .Append(mlContext.Regression.Trainers.FastTree());
            
            // Train model
            var model = pipeline.Fit(dataView);

                model.Transform(dataView));
            
            Console.WriteLine($"RÂ²: {metrics.RSquared:F2}");

            // Save model

        }
